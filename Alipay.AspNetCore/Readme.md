{

    {
        NotifyUrl = "",
        AppId = "",
        PrivateKey = "",
        AliPublicKey = ""
    };
    var request = new AlipayTradeAppPayModel
    {
        Body = "产品",
        Subject = "产品",
        OutTradeNo = "202007297777",
        TimeoutExpress = "15m",
        TotalAmount = $"{10:0.##}",

    };

    {
        Assert.Fail();
    }
    else
    {
        Assert.Pass();
    }
}
